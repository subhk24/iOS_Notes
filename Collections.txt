func printFullName(firstName: String, middleName: String?, lastName: String){
    if let unwrappedName = middleName{
        print("\(firstName) \(unwrappedName) \(lastName)")
    } else{
        print("\(firstName) \(lastName)")
    }
}
printFullName(firstName: "Subhdeep", middleName: nil, lastName: "Kaur")
printFullName(firstName: "Subh", middleName: "Deep", lastName: "Kaur")

//Failable initializers = initializers that fail in some cases (basically when the value jo hmme nhi chahiye to usko fai krvade)
struct Toddler{
    var name: String
    var monthsOld: Int
    init?(name:String, monthsOld: Int){ //init? is a failable initializer agr ? na lgate vo normal initalizer bnn jata
        if monthsOld < 12 || monthsOld > 36 {
            return nil //should return nill failable initializer me himme ek baar nil return krvana hi hota
        }
        self.name = name
        self.monthsOld = monthsOld
    }
}
//let toddler = Toddler(name: "Cutie", monthsOld: 6) //not a toddler cuz toddler's age is from
let toddler = Toddler(name: "Chris", monthsOld: 12)
print(toddler)

if let toddler = toddler{
    print("Toddler's name is \(toddler.name) and age is \(toddler.monthsOld)")
} else{
    print("That ain't a toddler")
}

struct Person{
    var name: String
    var residence: Residence?
}
struct Residence{
    var address: Address?
}
struct Address{
    var buildingName: String?
    var streetName: String?
    var apartmentNumber: String?
}
let address = Address(buildingName: "ABC", streetName: "XYZ", apartmentNumber: "123")
let residence = Residence(address: address)
let person = Person(name: "Subhdeep", residence: residence)
//print(person.residence?.address?.apartmentNumber) aese optional value aari thi to usko unwrap krna pdta issi liye we used if let optional binding
if let residence = person.residence, let address = residence.address, let apartmentNumber = address.apartmentNumber{ //nested if ki jgah aese likh skte
    print(apartmentNumber)
}
let anotherPerson = Person(name: "Shubham", residence: Residence(address: Address(buildingName: "PQR", streetName: "qwe", apartmentNumber: "673"))) //agr koi aur person ka bnana ho
//agr isko aur short krna ho we use optional chaining
//optional chaining me agr khi nil aara hai to vo uske bahar aa jayega and this is the most efficient way
if let apartmentNumber = person.residence?.address?.apartmentNumber{ //optional chaining
    print("They live in apartment number \(apartmentNumber)")
}
