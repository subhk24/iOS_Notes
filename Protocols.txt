//agr comparable use krna pehle equatable hona chahiye pehle then it'll be comparable
struct Employee: Equatable, Comparable{
    let firstName: String
    let lastName: String
    let jobTitle: String
    let phoneNumber: String
    //agr == ka comparison krna hai to we use Equatable that'll compare 2 instances
    static func == (lhs: Employee, rhs: Employee) -> Bool{
        return lhs.firstName == rhs.firstName && lhs.lastName == lhs.lastName && lhs.jobTitle == rhs.jobTitle && lhs.phoneNumber == rhs.phoneNumber
    }
    static func < (lhs: Employee, rhs: Employee) -> Bool{
        return lhs.lastName < rhs.lastName
    }
}
struct Company{
    let name: String
    let employees: [Employee]
}

//making ur own protocol
protocol FullyNamed{
    var fullName: String {get}
    func sayFullName() //jbb khudki protocol bnate hai to function me nhi likhna pdta ki kya krr ra
}
struct Person: FullyNamed{ //we are adopting the protocol here
    var firstName: String
    var lastName: String
    var fullName: String{
        return "\(firstName) \(lastName)"
    }
    func sayFullName() {
        print(fullName)
    }
}

//extension = extend a class

extension String{ //extending the String class of Swift
    static var name: String{
        return "New Data"
    }
    var details: String{
        return "\(self)"
    }
}
print(String.name)
let str = "Hi Swift"
print(str.details)
