struct Temp{
    var celsius: Double
    var fahrenheit: Double
    var kelvin: Double
    init(celsius: Double){ //stored property
        self.celsius = celsius
        fahrenheit = celsius * 1.8 + 32
        kelvin = celsius + 273.15
    }
    init(fahrenheit: Double){
        self.fahrenheit = fahrenheit
        celsius = (fahrenheit-32) / 1.8
        kelvin = celsius + 273.15
    }
    init(kelvin: Double){
        self.kelvin = kelvin
        celsius = kelvin - 273.15
        fahrenheit = celsius * 1.8 + 32
    }
}
let temp = Temp(celsius: 67.3)
print(temp.celsius)
print(temp.kelvin)
print(temp.fahrenheit)

// willSet and didSet


struct A{
    var num1: Double{
        willSet{
            //have access to newValue, yeh property change hone se pehle use hota hai mtlb abhi hone valli hai property abhi hui ni
            print("Value is going to change to \(newValue)")
            
        }
        didSet{
            //have access to oldValue, yeh property change hone ke baad use hota hai means change ho chuki hai property
            print("Value was \(oldValue) earlier")
        }
    }
}
var num = A(num1: 10)
num.num1 = 50

}
